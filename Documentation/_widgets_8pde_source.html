97class CircularGauge {
98 float cx, cy, radius;
99 float minVal, maxVal;
100 float value = 0;
101 String label, unit;
102 color gaugeColor, gaugeDim;
103 int majorTicks, minorTicks;
104 float startAngle = PI * 0.75;
105 float sweepAngle = PI * 1.5;
106
119 CircularGauge(float cx, float cy, float radius, String label, String unit,
120 float minVal, float maxVal, color gaugeColor, color gaugeDim) {
121 this.cx = cx; this.cy = cy; this.radius = radius;
122 this.label = label; this.unit = unit;
123 this.minVal = minVal; this.maxVal = maxVal;
124 this.gaugeColor = gaugeColor; this.gaugeDim = gaugeDim;
125 this.majorTicks = 6; this.minorTicks = 5;
126 }
127
129 void draw() {
130 pushMatrix();
131 translate(cx, cy);
132 float endAngle = startAngle + sweepAngle;
133
134 // Outer ring shadow + bezel
135 noFill();
136 stroke(0, 40); strokeWeight(3);
137 arc(0, 0, radius*2+6, radius*2+6, startAngle, endAngle);
138 stroke(COL_BORDER); strokeWeight(2);
139 arc(0, 0, radius*2+2, radius*2+2, startAngle, endAngle);
140
141 // Background arc
142 float arcD = radius * 1.7;
143 stroke(gaugeDim, 60); strokeWeight(10);
144 arc(0, 0, arcD, arcD, startAngle, endAngle);
145
146 // Value arc + glow
147 float fraction = constrain((value - minVal) / (maxVal - minVal), 0, 1);
148 float valueAngle = startAngle + sweepAngle * fraction;
149 stroke(gaugeColor); strokeWeight(10);
150 arc(0, 0, arcD, arcD, startAngle, valueAngle);
151 stroke(gaugeColor, 40); strokeWeight(18);
152 arc(0, 0, arcD, arcD, startAngle, valueAngle);
153
154 // Tick marks
155 float innerR = radius * 0.78, outerR = radius * 0.9, labelR = radius * 0.65;
156 int decPlaces = (maxVal <= 10) ? 1 : 0;
157 for (int i = 0; i <= majorTicks; i++) {
158 float frac = (float)i / majorTicks;
159 float tickAngle = startAngle + sweepAngle * frac;
160 float cosA = cos(tickAngle), sinA = sin(tickAngle);
161
162 stroke(COL_TEXT_DIM); strokeWeight(1.5);
163 line(cosA * innerR, sinA * innerR, cosA * outerR, sinA * outerR);
164
165 fill(COL_TEXT_DIM); noStroke();
166 textAlign(CENTER, CENTER); textSize(9);
167 text(nf(minVal + (maxVal - minVal) * frac, 0, decPlaces), cosA * labelR, sinA * labelR);
168
169 // Minor ticks
170 if (i < majorTicks) {
171 float minorStep = sweepAngle / majorTicks / minorTicks;
172 float minInR = radius * 0.83, minOutR = radius * 0.88;
173 stroke(COL_DIM, 120); strokeWeight(0.8);
174 for (int j = 1; j < minorTicks; j++) {
175 float ma = tickAngle + minorStep * j;
176 line(cos(ma) * minInR, sin(ma) * minInR, cos(ma) * minOutR, sin(ma) * minOutR);
177 }
178 }
179 }
180
181 // Needle
182 float needleAngle = startAngle + sweepAngle * fraction;
183 float needleLen = radius * 0.72;
184 float nx = cos(needleAngle) * needleLen, ny = sin(needleAngle) * needleLen;
185 stroke(0, 60); strokeWeight(3);
186 line(2, 2, nx+2, ny+2);
187 stroke(gaugeColor); strokeWeight(2.5);
188 line(0, 0, nx, ny);
189 fill(COL_PANEL_HEADER); stroke(gaugeColor); strokeWeight(1.5);
190 ellipse(0, 0, 12, 12);
191
192 // Digital readout
193 noStroke(); textAlign(CENTER, CENTER);
194 fill(gaugeColor);
195 textSize(radius * 0.32);
196 text(nf(value, 0, (maxVal <= 10) ? 3 : 2), 0, radius * 0.28);
197 fill(gaugeColor, 180); textSize(radius * 0.16);
198 text(unit, 0, radius * 0.48);
199 fill(COL_TEXT_DIM); textSize(radius * 0.13);
200 text(label, 0, radius * 0.63);
201
202 popMatrix();
203 }
204}

216class ScrollingGraph extends Widget {
217 String title = "Waveform";
218 boolean showVoltage = true, showCurrent = true, showPower = false;
219 float voltScale = 30.0, currScale = 5.0, powerScale = 150.0;
220
228 ScrollingGraph(float x, float y, float w, float h) {
229 super(x, y, w, h);
230 }
231
233 void draw() {
234 fill(COL_GRAPH_BG); stroke(COL_BORDER); strokeWeight(1);
235 rect(x, y, w, h, 4);
236
237 // Title bar
238 fill(COL_PANEL_HEADER); noStroke();
239 rect(x+1, y+1, w-2, 22, 3, 3, 0, 0);
240 fill(COL_TEXT); textAlign(LEFT, CENTER); textSize(11);
241 text(title, x + 8, y + 12);
242
243 // Legend
244 float lx = x + w - 200, ly = y + 12;
245 textSize(9);
246 if (showVoltage) { drawLegendItem(lx, ly, COL_VOLT, "Voltage"); lx += 55; }
247 if (showCurrent) { drawLegendItem(lx, ly, COL_CURR, "Current"); lx += 55; }
248 if (showPower) { drawLegendItem(lx, ly, COL_POWER, "Power"); }
249
250 float gx = x + 45, gy = y + 28, gw = w - 55, gh = h - 38;
251
252 // Grid
253 stroke(COL_GRID); strokeWeight(0.5);
254 for (int i = 0; i <= 5; i++) { float yy = gy + gh * i / 5.0; line(gx, yy, gx + gw, yy); }
255 for (int i = 0; i <= 10; i++) { float xx = gx + gw * i / 10.0; line(xx, gy, xx, gy + gh); }
256
257 // Y-axis labels
258 fill(COL_TEXT_DIM); textSize(8); textAlign(RIGHT, CENTER);
259 for (int i = 0; i <= 5; i++) {
260 float yy = gy + gh * i / 5.0;
261 text(nf(voltScale * (1.0 - (float)i/5.0), 0, 0), gx - 4, yy);
262 }
263
264 // X-axis time labels
265 textAlign(CENTER, TOP);
266 float totalTimeSec = psu.historyCount * psu.pollInterval / 1000.0;
267 for (int i = 0; i <= 4; i++) {
268 float xx = gx + gw * i / 4.0;
269 text(nf(-totalTimeSec + totalTimeSec * ((float)i / 4.0), 0, 0) + "s", xx, gy + gh + 2);
270 }
271
272 // Plot traces
273 if (psu.historyCount > 1) {
274 int start = (psu.historyIndex - psu.historyCount + psu.GRAPH_HISTORY) % psu.GRAPH_HISTORY;
275 float xScale = gw / (float)(psu.GRAPH_HISTORY - 1);
276 if (showVoltage) drawTrace(gx, gy, gw, gh, xScale, start, psu.historyV, voltScale, COL_VOLT, 1.5);
277 if (showCurrent) drawTrace(gx, gy, gw, gh, xScale, start, psu.historyA, currScale, COL_CURR, 1.5);
278 if (showPower) drawTrace(gx, gy, gw, gh, xScale, start, psu.historyW, powerScale, COL_POWER, 1.0);
279 }
280
281 // Border
282 noFill(); stroke(COL_BORDER); strokeWeight(1);
283 rect(gx, gy, gw, gh);
284 }
285
293 void drawLegendItem(float lx, float ly, color c, String label) {
294 fill(c); noStroke(); rect(lx, ly-4, 8, 8);
295 fill(COL_TEXT_DIM); textAlign(LEFT, CENTER);
296 text(label, lx+12, ly);
297 }

312 void drawTrace(float gx, float gy, float gw, float gh, float xScale, int start, float[] data, float scale, color c, float sw) {
313 noFill(); stroke(c); strokeWeight(sw);
314 beginShape();
315 for (int i = 0; i < psu.historyCount; i++) {
316 int idx = (start + i) % psu.GRAPH_HISTORY;
317 vertex(gx + xScale * i, constrain(gy + gh * (1.0 - data[idx] / scale), gy, gy+gh));
318 }
319 endShape();
320 }
