253 int calcChecksum(int[] data, int start, int end) {
254 int sum = 0;
255 for (int i = start; i < end; i++) {
256 sum += data[i];
257 }
258 return sum & 0xFF;
259 }

266 byte[] floatToLE(float val) {
267 ByteBuffer bb = ByteBuffer.allocate(4);
268 bb.order(ByteOrder.LITTLE_ENDIAN);
269 bb.putFloat(val);
270 return bb.array();
271 }

281 float leToFloat(int b0, int b1, int b2, int b3) {
282 ByteBuffer bb = ByteBuffer.allocate(4);
283 bb.order(ByteOrder.LITTLE_ENDIAN);
284 bb.put((byte)b0);
285 bb.put((byte)b1);
286 bb.put((byte)b2);
287 bb.put((byte)b3);
288 bb.flip();
289 return bb.getFloat();
290 }

302 void sendPacket(int cmdType, int register_, int[] data) {
303 if (serialPort == null) return;
304 int len = (data != null) ? data.length : 0;
305 int[] pkt = new int[4 + len + 1];
306 pkt[0] = HEADER_REQ;
307 pkt[1] = cmdType;
308 pkt[2] = register_;
309 pkt[3] = len;
310 if (data != null) {
311 for (int i = 0; i < len; i++) pkt[4 + i] = data[i];
312 }
313 pkt[pkt.length - 1] = calcChecksum(pkt, 2, pkt.length - 1);
314 byte[] out = new byte[pkt.length];
315 for (int i = 0; i < pkt.length; i++) out[i] = (byte)(pkt[i] & 0xFF);
316 serialPort.write(out);
317 }

447 void processSerialByte(int b) {
448 rxBuf[rxPos] = b & 0xFF;
449 switch (rxPos) {
450 case 0:
451 if (b == HEADER_RESP) rxPos = 1;
452 return;
453 case 1: rxPos = 2; return;
454 case 2: rxPos = 3; return;
455 case 3:
456 rxExpectedLen = b & 0xFF;
457 rxPos = 4;
458 return;
459 default:
460 rxPos++;
461 if (rxPos >= 4 + rxExpectedLen + 1) {
462 int expectedChk = calcChecksum(rxBuf, 2, rxPos - 1);
463 int actualChk = rxBuf[rxPos - 1];
464 if (expectedChk == actualChk) {
465 processResponsePacket(rxBuf, rxPos);
466 }
467 rxPos = 0;
468 }
469 if (rxPos >= 500) rxPos = 0;
470 return;
471 }
472 }

487 void processResponsePacket(int[] buf, int len) {
488 int reg = buf[2];
489 int dataLen = buf[3];
490
491 if (reg == REG_LIVE_VALUES && dataLen >= 12) {
492 liveVoltage = leToFloat(buf[4], buf[5], buf[6], buf[7]);
493 liveCurrent = leToFloat(buf[8], buf[9], buf[10], buf[11]);
494 livePower = leToFloat(buf[12], buf[13], buf[14], buf[15]);
495 addHistorySample();
496 }
497 else if (reg == REG_INPUT_VOLT && dataLen >= 4) {
498 inputVoltage = leToFloat(buf[4], buf[5], buf[6], buf[7]);
499 }
500 else if (reg == REG_SET_VOLT && dataLen >= 4) {
501 setVoltage = leToFloat(buf[4], buf[5], buf[6], buf[7]);
502 if (!gotFirstSetpoint) { gotFirstSetpoint = true; onSetpointsReceived(); }
503 }
504 else if (reg == REG_SET_CURR && dataLen >= 4) {
505 setCurrent = leToFloat(buf[4], buf[5], buf[6], buf[7]);
506 if (!gotFirstSetpoint) { gotFirstSetpoint = true; onSetpointsReceived(); }
507 }
508 else if (reg == REG_TEMPERATURE && dataLen >= 4) {
509 temperature = leToFloat(buf[4], buf[5], buf[6], buf[7]);
510 }
511 else if (reg == REG_MAX_VOLT && dataLen >= 4) {
512 maxVoltage = leToFloat(buf[4], buf[5], buf[6], buf[7]);
513 }
514 else if (reg == REG_MAX_CURR && dataLen >= 4) {
515 maxCurrent = leToFloat(buf[4], buf[5], buf[6], buf[7]);
516 }
517 else if (reg == REG_MODEL_NAME && dataLen >= 4) {
518 StringBuilder sb = new StringBuilder();
519 for (int i = 0; i < dataLen; i++) sb.append((char)buf[4+i]);
520 deviceId = sb.toString();
521 }
522 else if (reg == REG_OUTPUT && dataLen >= 1) {
523 outputOn = (buf[4] == 1);
524 }
525 else if (reg == REG_MODE && dataLen >= 1) {
526 outputMode = buf[4];
527 }
528 else if (reg == REG_PROTECTION && dataLen >= 1) {
529 protectionStatus = buf[4];
530 }
531 else if (reg == REG_BRIGHTNESS && dataLen >= 1) {
532 brightness = buf[4];
533 }
534 else if (reg == REG_OVP && dataLen >= 4) { ovpLimit = leToFloat(buf[4], buf[5], buf[6], buf[7]); }
535 else if (reg == REG_OCP && dataLen >= 4) { ocpLimit = leToFloat(buf[4], buf[5], buf[6], buf[7]); }
536 else if (reg == REG_OPP && dataLen >= 4) { oppLimit = leToFloat(buf[4], buf[5], buf[6], buf[7]); }
537 else if (reg == REG_OTP && dataLen >= 4) { otpLimit = leToFloat(buf[4], buf[5], buf[6], buf[7]); }
538 else if (reg == REG_CAP_AH && dataLen >= 4) { capacityAh = leToFloat(buf[4], buf[5], buf[6], buf[7]); }
539 else if (reg == REG_CAP_WH && dataLen >= 4) { capacityWh = leToFloat(buf[4], buf[5], buf[6], buf[7]); }
540 else if (reg == REG_ALL && dataLen >= 139) {
541 // Bulk dump — offsets verified from hardware testing
542 inputVoltage = leToFloat(buf[4+0], buf[4+1], buf[4+2], buf[4+3]);
543 setVoltage = leToFloat(buf[4+4], buf[4+5], buf[4+6], buf[4+7]);
544 setCurrent = leToFloat(buf[4+8], buf[4+9], buf[4+10], buf[4+11]);
545 liveVoltage = leToFloat(buf[4+12], buf[4+13], buf[4+14], buf[4+15]);
546 liveCurrent = leToFloat(buf[4+16], buf[4+17], buf[4+18], buf[4+19]);
547 livePower = leToFloat(buf[4+20], buf[4+21], buf[4+22], buf[4+23]);
548 temperature = leToFloat(buf[4+24], buf[4+25], buf[4+26], buf[4+27]);
549 // Presets at offset 28 (6 pairs of V+A floats = 48 bytes)
550 for (int i = 0; i < 6; i++) {
551 int off = 4 + 28 + i * 8;
552 presetV[i] = leToFloat(buf[off], buf[off+1], buf[off+2], buf[off+3]);
553 presetA[i] = leToFloat(buf[off+4], buf[off+5], buf[off+6], buf[off+7]);
554 }
555 maxVoltage = leToFloat(buf[4+76], buf[4+77], buf[4+78], buf[4+79]);
556 maxCurrent = leToFloat(buf[4+80], buf[4+81], buf[4+82], buf[4+83]);
557 ovpLimit = leToFloat(buf[4+84], buf[4+85], buf[4+86], buf[4+87]);
558 ocpLimit = leToFloat(buf[4+88], buf[4+89], buf[4+90], buf[4+91]);
559 oppLimit = leToFloat(buf[4+92], buf[4+93], buf[4+94], buf[4+95]);
560 otpLimit = leToFloat(buf[4+96], buf[4+97], buf[4+98], buf[4+99]);
561 addHistorySample();
562 onSetpointsReceived();
563 }
564
565 // Preset registers (individual reads)
566 for (int i = 0; i < 6; i++) {
567 if (reg == REG_PRESET_V[i] && dataLen >= 4) {
568 presetV[i] = leToFloat(buf[4], buf[5], buf[6], buf[7]);
569 }
570 if (reg == REG_PRESET_A[i] && dataLen >= 4) {
571 presetA[i] = leToFloat(buf[4], buf[5], buf[6], buf[7]);
572 }
573 }
574 }

602 boolean connectToPort(String portName) {
603 try {
604 serialPort = new Serial((PApplet)DPS150_PSU.this, portName, 115200);
605 serialPort.buffer(1);
606 connectedPortName = portName;
607 delay(100);
608 sendConnect();
609 delay(500);
610 connected = true;
611 // Request setpoints and state (staggered so PSU can process each)
612 sendReadRegister(REG_SET_VOLT);
613 delay(50);
614 sendReadRegister(REG_SET_CURR);
615 delay(50);
616 sendReadRegister(REG_OUTPUT);
617 delay(50);
618 sendReadRegister(REG_MODE);
619 delay(50);
620 sendReadRegister(REG_PROTECTION);
621 delay(50);
622 sendReadRegister(REG_ALL);
623 return true;
624 } catch (Exception e) {
625 println("Connection failed: " + e.getMessage());
626 serialPort = null;
627 connected = false;
628 return false;
629 }
630 }

661 void pollPSU() {
662 if (!connected || serialPort == null) return;
663 long now = millis();
664 if (now - lastPollTime >= pollInterval) {
665 lastPollTime = now;
666 sendReadLive();
667 pollCycle++;
668 if (pollCycle % 5 == 0) {
669 sendReadRegister(REG_MODE);
670 sendReadRegister(REG_PROTECTION);
671 }
