444void drawSweepPreview(float x, float y, float w, float h, float startVal, float endVal, float stepVal, float currentVal, boolean isActive) {
445 fill(COL_GRAPH_BG); stroke(COL_BORDER); strokeWeight(1);
446 rect(x, y, w, h, 3);
447 fill(COL_TEXT_DIM); textAlign(LEFT, TOP); textSize(9);
448 text("Preview", x + 5, y + 3);
449
450 float gx = x + 35, gy = y + 18, gw = w - 45, gh = h - 28;
451
452 stroke(COL_GRID); strokeWeight(0.5);
453 for (int i = 0; i <= 4; i++) { float yy = gy + gh * i / 4.0; line(gx, yy, gx + gw, yy); }
454
455 if (stepVal <= 0) stepVal = 0.01;
456 int numSteps = min((int)(abs(endVal - startVal) / stepVal) + 1, 200);
457 float minV = min(startVal, endVal), maxV = max(startVal, endVal);
458 if (maxV <= minV) maxV = minV + 1;
459 float range = maxV - minV;
460
461 fill(COL_TEXT_DIM); textAlign(RIGHT, CENTER); textSize(8);
462 for (int i = 0; i <= 4; i++) {
463 float yy = gy + gh * i / 4.0;
464 text(nf(maxV - range * ((float)i / 4.0), 0, 1), gx - 3, yy);
465 }
466
467 // Staircase
468 stroke(COL_ACCENT_LITE); strokeWeight(1.5); noFill();
469 boolean goingUp = (endVal >= startVal);
470 beginShape();
471 for (int i = 0; i < numSteps; i++) {
472 float val = constrain(goingUp ? startVal + i * stepVal : startVal - i * stepVal, minV, maxV);
473 float py = gy + gh * (1.0 - (val - minV) / range);
474 vertex(gx + gw * ((float)i / numSteps), py);
475 vertex(gx + gw * ((float)(i + 1) / numSteps), py);
476 }
477 endShape();
478
479 // Position marker
480 if (isActive && advState == ADV_RUNNING) {
481 float markerY = constrain(gy + gh * (1.0 - (currentVal - minV) / range), gy, gy + gh);
482 fill(COL_ON); noStroke(); ellipse(gx + gw / 2, markerY, 8, 8);
483 fill(COL_ON, 40); ellipse(gx + gw / 2, markerY, 16, 16);
484 }
485}

494void updateAdvanced() {
495 if (advState != ADV_RUNNING || !psu.connected) return;
496 long now = millis();
497 switch (advMode) {
498 case 0: updateSequentialOutput(now); break;
499 case 1: case 2: updateSweep(now); break;
500 }
501}
502
508void updateSequentialOutput(long now) {
509 if (seqCurrentStep < 0) {
510 seqCurrentStep = nextEnabledStep(-1);
511 if (seqCurrentStep < 0) { advStop(); return; }
512 applySequentialStep(seqCurrentStep);
513 return;
514 }
515 if ((now - seqStepStartTime) >= (long)(seqDelay[seqCurrentStep] * 1000)) {
516 seqStatus[seqCurrentStep] = 2;
517 int nextStep = nextEnabledStep(seqCurrentStep);
518 if (nextStep < 0) {
519 seqCurrentLoop++;
520 if (seqLoopCount > 0 && seqCurrentLoop >= seqLoopCount) {
521 advStop(); setStatus("Sequence completed. " + seqCurrentLoop + " loop(s) done."); return;
522 }
523 resetSeqStatus();
524 seqCurrentStep = nextEnabledStep(-1);
525 if (seqCurrentStep < 0) { advStop(); return; }
526 } else {
527 seqCurrentStep = nextStep;
528 }
529 applySequentialStep(seqCurrentStep);
530 }
531}
532
534void updateSweep(long now) {
535 boolean isVoltage = (advMode == 1);
536 float delayMs = (isVoltage ? vsDelay : csDelay) * 1000;
537 if ((now - advLastStepTime) < (long) delayMs) return;
538 advLastStepTime = now;
539
540 float curVal = isVoltage ? vsCurrentValue : csCurrentValue;
541 float startV = isVoltage ? vsStartVoltage : csStartCurrent;
542 float endV = isVoltage ? vsEndVoltage : csEndCurrent;
543 float stepV = isVoltage ? vsStepVoltage : csStepCurrent;
544
545 if (isVoltage) { psu.sendSetVoltage(curVal); psu.sendSetCurrent(vsFixedCurrent); }
546 else { psu.sendSetVoltage(csFixedVoltage); psu.sendSetCurrent(curVal); }
547 if (!psu.outputOn) { psu.sendOutputOn(); psu.outputOn = true; }
548 setStatus((isVoltage ? "V Sweep: " + nf(curVal, 0, 3) + "V" : "I Sweep: " + nf(curVal, 0, 3) + "A"));
549
550 float nextVal = (startV <= endV) ? curVal + stepV : curVal - stepV;
551 boolean done = (startV <= endV) ? (nextVal > endV + 0.0001) : (nextVal < endV - 0.0001);
552 if (done) {
553 advStop();
554 setStatus((isVoltage ? "Voltage" : "Current") + " sweep completed.");
555 } else {
556 if (isVoltage) vsCurrentValue = nextVal; else csCurrentValue = nextVal;
557 }
558}

565void applySequentialStep(int step) {
566 seqStatus[step] = 1;
567 seqStepStartTime = millis();
568 psu.sendSetVoltage(seqVoltage[step]);
569 psu.sendSetCurrent(seqCurrent[step]);
570 if (!psu.outputOn) { psu.sendOutputOn(); psu.outputOn = true; }
571 setStatus("Seq step " + (step+1) + ": " + nf(seqVoltage[step],0,3) + "V / " + nf(seqCurrent[step],0,3) + "A");
572}

580int nextEnabledStep(int after) {
581 for (int i = after + 1; i < SEQ_MAX_ROWS; i++) if (seqEnabled[i]) return i;
582 return -1;
583}

590int countEnabledSteps() {

608void advStart() {
609 if (!psu.connected) return;
610 advState = ADV_RUNNING;
611 switch (advMode) {
612 case 0: resetSeqStatus(); seqCurrentStep = -1; seqCurrentLoop = 0; setStatus("Sequential output started."); break;
613 case 1: vsCurrentValue = vsStartVoltage; advLastStepTime = millis() - (long)(vsDelay * 1000); setStatus("Voltage sweep started."); break;
614 case 2: csCurrentValue = csStartCurrent; advLastStepTime = millis() - (long)(csDelay * 1000); setStatus("Current sweep started."); break;
615 }
616}
617
621void advPause() { advState = ADV_PAUSED; setStatus("Advanced output paused."); }
622
626void advContinue() {
627 if (advState != ADV_PAUSED) return;
628 advState = ADV_RUNNING;
629 if (advMode == 0 && seqCurrentStep >= 0) seqStepStartTime = millis();
630 else advLastStepTime = millis();
631 setStatus("Advanced output resumed.");
632}
633
637void advStop() {
638 advState = ADV_IDLE;
639 seqCurrentStep = -1;
640 if (advMode == 0) resetSeqStatus();
641 setStatus("Advanced output stopped.");
642}
643
647void advSingleStep() {
648 if (advMode != 0) return;
649 if (advState == ADV_IDLE) {
650 seqCurrentStep = nextEnabledStep(-1);
651 if (seqCurrentStep < 0) return;
652 seqCurrentLoop = 0; resetSeqStatus();
653 } else if (advState == ADV_PAUSED) {
654 seqStatus[seqCurrentStep] = 2;
655 int next = nextEnabledStep(seqCurrentStep);
656 if (next < 0) { advStop(); setStatus("Sequence completed (single step)."); return; }
657 seqCurrentStep = next;
658 }
659 advState = ADV_PAUSED;
660 applySequentialStep(seqCurrentStep);
661 setStatus("Single step: " + (seqCurrentStep + 1));
662}

671void handleAdvancedClick() {
672 if (!advancedOpen) return;
673
674 if (btnAdvClose.clicked()) {

677 return;
678 }

683 if (btnAdvModeCS.clicked()) advMode = 2;
684 }
685
686 if (btnLoopUp.clicked()) seqLoopCount = min(seqLoopCount + 1, 999);
687 if (btnLoopDown.clicked()) seqLoopCount = max(seqLoopCount - 1, 0);
688
689 if (btnAdvStart.clicked()) advStart();
690 if (btnAdvPause.clicked()) advPause();
691 if (btnAdvContinue.clicked()) advContinue();
692 if (btnAdvStop.clicked()) advStop();
693 if (btnAdvSingleStep.clicked()) advSingleStep();

696 for (int i = 0; i < SEQ_MAX_ROWS; i++) {
697 seqVoltage[i] = 0; seqCurrent[i] = 0; seqDelay[i] = 1.0;
698 seqEnabled[i] = false; seqStatus[i] = 0;
699 }
700 seqEnabled[0] = true;
701 vsEditField = -1; csEditField = -1; seqEditRow = -1; seqEditCol = -1;
702 }

834String editNumericBuffer(String buf, char k) {
835 if (k == BACKSPACE || k == DELETE) {
836 return buf.length() > 0 ? buf.substring(0, buf.length()-1) : buf;
837 } else if ((k >= '0' && k <= '9') || k == '.') {
838 if (k == '.' && buf.indexOf('.') >= 0) return buf; // already has decimal
839 if (buf.length() < 10) return buf + k;
840 }
841 return null; // not a numeric key
842}

850void handleAdvancedKey(char k, int kCode) {
851 if (!advancedOpen) return;
852
853 // Sequential table editing
854 if (advMode == 0 && seqEditRow >= 0 && seqEditCol >= 0) {
855 String result = editNumericBuffer(seqEditBuffer, k);
856 if (result != null) { seqEditBuffer = result; return; }
857 if (k == ENTER || k == RETURN) { commitSeqEdit(); seqEditRow = -1; seqEditCol = -1; }
858 else if (k == TAB) {
859 commitSeqEdit();
860 seqEditCol++;
861 if (seqEditCol > 2) { seqEditCol = 0; seqEditRow = (seqEditRow + 1) % SEQ_MAX_ROWS; }
862 switch (seqEditCol) {
863 case 0: seqEditBuffer = nf(seqVoltage[seqEditRow], 0, 3); break;
864 case 1: seqEditBuffer = nf(seqCurrent[seqEditRow], 0, 3); break;
865 case 2: seqEditBuffer = nf(seqDelay[seqEditRow], 0, 1); break;
866 }
867 }
868 return;
869 }
870
871 // Sweep field editing (unified for V and I sweep)
872 int editField = (advMode == 1) ? vsEditField : (advMode == 2) ? csEditField : -1;
873 if (editField >= 0) {
874 String buf = (advMode == 1) ? vsEditBuffer : csEditBuffer;
875 String result = editNumericBuffer(buf, k);
876 if (result != null) {
877 if (advMode == 1) vsEditBuffer = result; else csEditBuffer = result;
878 return;
879 }
880 if (k == ENTER || k == RETURN || k == TAB) commitSweepEdit(advMode);
881 return;
882 }
883
884 if (k == ESC) {
885 key = 0;
886 if (advState != ADV_IDLE) advStop();
887 advancedOpen = false;
888 }
