98void applyTheme(Controller c, int bg, int fg, int active) {

308void refreshPorts() {
309 availablePorts = Serial.list();
310 selectedPortIndex = 0;
311 for (int i = 0; i < availablePorts.length; i++) {
312 if (availablePorts[i].contains("cu.usbmodem14798A3C")) {
313 selectedPortIndex = i;
314 break;
315 }
316 }
317 selectedPortName = (availablePorts.length > 0) ? availablePorts[selectedPortIndex] : "";
318}

328void drawGUI() {
329 background(COL_BG);
330
331 // ---- TOP BAR ----
332 fill(COL_PANEL);
333 noStroke();
334 rect(0, 0, WIN_W, TOP_BAR_H);
335 stroke(COL_BORDER);
336 strokeWeight(1);
337 line(0, TOP_BAR_H, WIN_W, TOP_BAR_H);
338
339 // Connection LED
340 float ledPulse = psu.connected ? (sin(millis() * 0.005) * 0.3 + 0.7) : 0.3;
341 fill(psu.connected ? color(COL_ON, (int)(255*ledPulse)) : #661111);
342 noStroke();
343 ellipse(14, 20, 10, 10);
344 if (psu.connected) {
345 fill(COL_ON, 30);
346 ellipse(14, 20, 20, 20);
347 }
348
349 // Port display
350 fill(COL_TEXT);
351 textAlign(LEFT, CENTER);
352 textSize(11);
353 if (psu.connected) {
354 text("Connected: " + psu.connectedPortName, 26, 20);
355 } else {
356 text("Port:", 26, 20);
357 fill(COL_INPUT_BG);
358 stroke(COL_INPUT_BORDER);
359 rect(206, 8, 350, 24, 3);
360 fill(COL_TEXT);
361 noStroke();
362 textAlign(CENTER, CENTER);
363 textSize(10);
364 text((availablePorts.length > 0) ? availablePorts[selectedPortIndex] : "(no ports found)", 381, 20);
365 }
366
367 // Top bar visibility — only update when connection state changes
368 boolean conn = psu.connected;
369 if (conn != prevConnState) {
370 prevConnState = conn;
371 cPortPrev.setVisible(!conn);
372 cPortNext.setVisible(!conn);
373 cRefreshPorts.setVisible(!conn);
374 cOpenAdvanced.setVisible(conn);
375 if (conn) {
376 cConnToggle.setLabel("Disconnect");
377 applyRedTheme(cConnToggle);
378 grpConnected.show();
379 } else {
380 cConnToggle.setLabel("Connect");
381 applyGreenTheme(cConnToggle);
382 grpConnected.hide();
383 }
384 }
385 cConnToggle.setLock(!conn && availablePorts.length == 0);
386
387 if (!conn) {
388 fill(COL_TEXT_DIM);
389 textAlign(CENTER, CENTER);
390 textSize(18);
391 text("FNIRSI DPS-150 Power Supply Control", WIN_W / 2, WIN_H / 2 - 30);
392 textSize(12);
393 text("Select a serial port and click Connect", WIN_W / 2, WIN_H / 2 + 10);
394 }
395
396 if (conn) {
397 drawConnectedGUI();
398 }
399
400 // ---- STATUS BAR ----
401 fill(COL_PANEL);
402 noStroke();
403 rect(0, WIN_H - 22, WIN_W, 22);
404 stroke(COL_BORDER);
405 line(0, WIN_H - 22, WIN_W, WIN_H - 22);
406 fill(COL_TEXT_DIM);
407 textAlign(LEFT, CENTER);
408 textSize(10);
409 text(statusMessage, 8, WIN_H - 11);
410 textAlign(RIGHT, CENTER);
411 text("DPS-150 Control | " + nf(frameRate, 0, 0) + " fps", WIN_W - 10, WIN_H - 11);
412}

582void handleCp5Event(ControlEvent e) {
583 String name = e.getController().getName();
584
585 // --- Top bar ---
586 if (name.equals("btnPortPrev")) {
587 if (!psu.connected && selectedPortIndex > 0) {
588 selectedPortIndex--;
589 selectedPortName = availablePorts[selectedPortIndex];
590 }
591 }
592 else if (name.equals("btnPortNext")) {
593 if (!psu.connected && selectedPortIndex < availablePorts.length - 1) {
594 selectedPortIndex++;
595 selectedPortName = availablePorts[selectedPortIndex];
596 }
597 }
598 else if (name.equals("btnRefreshPorts")) {
599 refreshPorts();
600 setStatus("Ports refreshed. Found " + availablePorts.length + " port(s).");
601 }
602 else if (name.equals("btnConnToggle")) {
603 if (psu.connected) {
604 psu.disconnectFromPSU();
605 setpointsPopulated = false;
606 setStatus("Disconnected.");
607 } else if (availablePorts.length > 0) {
608 setStatus("Connecting to " + availablePorts[selectedPortIndex] + "...");
609 if (psu.connectToPort(availablePorts[selectedPortIndex])) {
610 setStatus("Connected to " + psu.connectedPortName + " — Set V/A values and click Apply.");
611 } else {
612 setStatus("Connection failed!");
613 }
614 }
615 }
616 else if (name.equals("btnOpenAdvanced")) { advancedOpen = true; }
617
618 // --- Output toggle ---
619 else if (name.equals("btnOutput")) {
620 if (psu.connected) {
621 boolean newState = ((Toggle) e.getController()).getState();
622 if (newState) { psu.sendOutputOn(); psu.outputOn = true; setStatus("Output ON"); }
623 else { psu.sendOutputOff(); psu.outputOn = false; setStatus("Output OFF"); }
624 outputToggleTime = millis();
625 }
626 }
627
628 // --- Voltage / Current adjust ---
629 else if (name.equals("btnVoltUp")) adjustTf(cTfSetV, 0.1, 0, 30);
630 else if (name.equals("btnVoltDown")) adjustTf(cTfSetV, -0.1, 0, 30);
631 else if (name.equals("btnVoltUpFine")) adjustTf(cTfSetV, 0.01, 0, 30);
632 else if (name.equals("btnVoltDownFine")) adjustTf(cTfSetV, -0.01, 0, 30);
633 else if (name.equals("btnCurrUp")) adjustTf(cTfSetA, 0.1, 0, 5);
634 else if (name.equals("btnCurrDown")) adjustTf(cTfSetA, -0.1, 0, 5);
635 else if (name.equals("btnCurrUpFine")) adjustTf(cTfSetA, 0.01, 0, 5);
636 else if (name.equals("btnCurrDownFine")) adjustTf(cTfSetA, -0.01, 0, 5);
637
638 // --- Apply / textfield submit ---
639 else if (name.equals("btnApply") || name.equals("tfSetVoltage") || name.equals("tfSetCurrent")) {
640 applySetpoints();
641 }
642
643 // --- Presets ---
644 else if (name.startsWith("btnPresetLoad")) {
645 int i = name.charAt(name.length()-1) - '0';
646 if (psu.connected) {
647 psu.sendLoadPreset(i);
648 cTfSetV.setText(nf(psu.presetV[i], 0, 3));
649 cTfSetA.setText(nf(psu.presetA[i], 0, 3));
650 setStatus("Loaded preset " + (i+1));
651 }
652 }
653 else if (name.startsWith("btnPresetSave")) {
654 int i = name.charAt(name.length()-1) - '0';
655 if (psu.connected) {
656 float v = parseTf(cTfSetV), a = parseTf(cTfSetA);
657 psu.sendSavePreset(i, v, a);
658 psu.presetV[i] = v;
659 psu.presetA[i] = a;
660 setStatus("Saved preset " + (i+1) + ": " + nf(v,0,2) + "V / " + nf(a,0,2) + "A");
661 }
662 }
663
664 // --- Info refresh ---
665 else if (name.equals("btnRefreshAll")) {
666 if (psu.connected) {
667 psu.gotFirstSetpoint = false;
668 setpointsPopulated = false;
669 psu.sendReadRegister(REG_ALL);
670 setStatus("Refreshing all parameters...");
671 }
672 }
673
674 // --- Protection ---
675 else if (name.equals("tfOVP") || name.equals("tfOCP") || name.equals("tfOPP") || name.equals("tfOTP") || name.equals("btnApplyProtection")) {
676 applyProtection();
677 }
678
679 // --- Brightness ---
680 else if (name.equals("sliderBrightness")) { /* wait for Set click */ }
681 else if (name.equals("btnApplyBrightness")) {
682 if (psu.connected) {
683 int brt = (int) cBrightness.getValue();
684 psu.brightness = brt;
685 psu.sendSetBrightness(brt);
686 setStatus("Brightness set to " + brt);
687 }
688 }
689
690 // --- Vset / Iset sliders (only when user is actively dragging) ---
691 else if (name.equals("sliderVset")) {
692 if (psu.connected && sliderVset.isMouseOver()) {
693 float v = sliderVset.getValue();
694 psu.sendSetVoltage(v);
695 psu.setVoltage = v;
696 cTfSetV.setText(nf(v, 0, 3));
697 setStatus("Voltage set to " + nf(v, 0, 3) + "V");
698 }
699 }
700 else if (name.equals("sliderIset")) {
701 if (psu.connected && sliderIset.isMouseOver()) {
702 float a = sliderIset.getValue();
703 psu.sendSetCurrent(a);
704 psu.setCurrent = a;
705 cTfSetA.setText(nf(a, 0, 3));
706 setStatus("Current set to " + nf(a, 0, 3) + "A");
707 }
708 }
709
710 // --- Graph toggles ---
711 else if (name.equals("btnGraphV")) { graph.showVoltage = !graph.showVoltage; }
712 else if (name.equals("btnGraphA")) { graph.showCurrent = !graph.showCurrent; }
713 else if (name.equals("btnGraphW")) { graph.showPower = !graph.showPower; }
714
715 // --- Logging ---
716 else if (name.equals("btnStartLog")) {
717 if (psu.connected && !psu.logging) { psu.startLogging(); setStatus("Logging started: " + psu.logFileName); }
718 }
719 else if (name.equals("btnStopLog")) {
720 if (psu.logging) { psu.stopLogging(); setStatus("Logging stopped. " + psu.logSampleCount + " samples saved."); }
721 }
722}

752void applySetpoints() {
753 if (!psu.connected) return;
754 float v = constrain(parseTf(cTfSetV), 0, psu.maxVoltage);
755 float a = parseTf(cTfSetA);
756 if (a < 0.001) a = psu.setCurrent;
757 a = constrain(a, 0, psu.maxCurrent);
758 psu.sendSetVoltage(v);
759 delay(100);
760 psu.sendSetCurrent(a);
761 psu.setVoltage = v;
762 psu.setCurrent = a;
763 cTfSetV.setText(nf(v, 0, 3));
764 cTfSetA.setText(nf(a, 0, 3));
765 setStatus("Applied: " + nf(v,0,3) + "V / " + nf(a,0,3) + "A");
766}

771void applyProtection() {
772 if (!psu.connected) return;
773 psu.sendSetOVP(parseTf(cTfOVP));
774 psu.sendSetOCP(parseTf(cTfOCP));
775 psu.sendSetOPP(parseTf(cTfOPP));
776 psu.sendSetOTP(parseTf(cTfOTP));
777 setStatus("Protection limits applied.");
778}
